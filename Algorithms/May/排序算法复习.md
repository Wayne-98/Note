package sort;

import java.util.Arrays;
import java.util.Random;

public class Sort {
    public static void main(String[] args) {
        int[] a = new int[20];
        Random r = new Random();
        for (int i = 0; i < a.length; i++) {
            a[i] = r.nextInt(10);
        }
//        quickSort(a, 0, a.length - 1);
        mergeSort(a, 0, a.length - 1);
        System.out.println(Arrays.toString(a));
    }

    static void mergeSort(int[] nums, int left, int right) {
        if (left >= right) return;
        int mid = left + ((right - left) >> 1);
        mergeSort(nums, left, mid);
        mergeSort(nums, mid + 1, right);
        merge(nums, left, mid, right);
    }

    static void merge(int[] nums, int left, int mid, int right) {
        int[] t = new int[right - left + 1];
        int index = 0;
        int L = left, m = mid + 1;
        while (L <= mid && m <= right) {
            if (nums[L] < nums[m]) t[index++] = nums[L++];
            else t[index++] = nums[m++];
        }
        while (L <= mid) t[index++] = nums[L++];
        while (m <= right) t[index++] = nums[m++];
        for (int i = 0; i < t.length; i++) {
            nums[left + i] = t[i];
        }
    }


    static void quickSort(int[] nums, int left, int right) {
        if (left >= right) return;
        int partition = partition(nums, left, right);
        quickSort(nums, left, partition - 1);
        quickSort(nums, partition + 1, right);
    }

    static int partition(int[] nums, int left, int right) {
        Random r = new Random();
        int t = r.nextInt(right - left);
        swap(nums, left, left + t);
        int pivot = nums[left];
        int L = left, R = right;
        while (L < R) {
            while (L < R && nums[R] >= pivot) R--;
            nums[L] = nums[R];
            while (L < R && nums[L] <= pivot) L++;
            nums[R] = nums[L];
        }
        nums[L] = pivot;
        return L;
    }

    static void bubbleSort(int[] nums) {
        for (int i = 0; i < nums.length - 1; i++) {
            boolean flag = false;
            for (int j = nums.length - 1; j > i; j--) {
                if (nums[j] < nums[j - 1]) {
                    swap(nums, j, j - 1);
                    flag = true;
                }
            }
            if (!flag) break;
        }
    }

    static void shellSort(int[] nums) {
        int len = nums.length;
        int dk = 3;
        int h = 1;
        while (h < len / dk) {
            h = dk * h + 1;
        }

        while (h >= 1) {
            for (int i = h; i < len; i++) {
                for (int j = i; j >= h && nums[j] < nums[j - h]; j -= h) {
                    swap(nums, j, j - h);
                }
            }
            h /= dk;
        }
    }

    static void heapSort(int[] nums) {
        buildMaxHeap(nums);
        for (int i = nums.length - 1; i >= 0; i--) {
            swap(nums, 0, i);
            adjustDown(nums, 0, i);
        }
    }


    static void buildMaxHeap(int[] nums) {
        for (int i = nums.length / 2; i >= 0; i--) {
            adjustDown(nums, i, nums.length);
        }
    }

    static void adjustDown(int[] nums, int k, int len) {
        int t = nums[k];
        for (int i = k * 2; i < len; i *= 2) {
            if (i < len - 1 && nums[i + 1] > nums[i]) i++;
            if (nums[i] <= t) break;
            else {
                nums[k] = nums[i];
                k = i;
            }
        }
        nums[k] = t;
    }

    static void binaryInsertSort(int[] nums) {
        if (nums == null || nums.length < 2) return;
        for (int i = 1; i < nums.length; i++) {
            int t = nums[i];
            int L = 0, R = i - 1;
            while (L <= R) {
                int mid = L + ((R - L) >> 1);
                if (nums[mid] > t) R = mid - 1;
                else L = mid + 1;
            }

            for (int j = i - 1; j > R; j--)
                nums[j + 1] = nums[j];
            nums[R + 1] = t;
        }
    }

    static void insertSort2(int[] nums) {
        if (nums == null || nums.length < 2) return;
        for (int i = 1; i < nums.length; i++) {
            for (int j = i - 1; j >= 0 && nums[j] > nums[j + 1]; j--)
                swap(nums, j, j + 1);
        }
    }

    static void insertSort(int[] nums) {
        if (nums == null || nums.length < 2) return;
        for (int i = 1; i < nums.length; i++) {
            int t = nums[i];
            int j;
            for (j = i - 1; j >= 0 && nums[j] > t; j--)
                nums[j + 1] = nums[j];
            nums[j + 1] = t;
        }
    }

    static void swap(int[] nums, int i, int j) {
        int t = nums[i];
        nums[i] = nums[j];
        nums[j] = t;
    }
}
